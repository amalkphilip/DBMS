create or replace function factorial(n int)
return int is
  fact int:=1;
begin
  for i in 1..n loop
    fact:=fact*i;
  end loop;
  return fact;
end;
/
declare 
  result int;
begin
  result:=factorial(5);
  DBMS_OUTPUT.PUT_LINE('factorial is '||result);
end;
/
